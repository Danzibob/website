function setup(){
    createCanvas(600,400)
    colorMode(HSB)
}
function draw(){
    hue = map(mouseX, 0, width, 0, 360)
    brightness = map(mouseY, 0, height, 0, 100)
    fill(hue, 100, brightness)
    background(100)
    ellipse(mouseX, mouseY, 60, 60)
}
        

var x
var y
function setup(){
    createCanvas(600,400)
    background(51)
    x = width/2 -10
    y = height/2 -10
    rect(x, y, 20,20)
    noStroke()
}
function draw(){
    if(keyIsDown(LEFT_ARROW)){x-=5}
    if(keyIsDown(RIGHT_ARROW)){x+=5}
    if(keyIsDown(UP_ARROW)){y-=5}
    if(keyIsDown(DOWN_ARROW)){y+=5}
    rect(x, y, 20,20)
}
